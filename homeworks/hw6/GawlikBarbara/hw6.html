def czarny_krol_plansza(data, title):

    counts = data.pivot_table(
        index="black-king-rank",
        columns="black-king-file",
        values="white-depth-of-win",
        aggfunc="count",
        fill_value=0
    )

    counts = counts.reindex(index=range(8), columns=range(8), fill_value=0)
    # eby nie byo pustych miejsc

    kolumny = ['A', 'B', 'C', 'D', 'E', 'F', 'G', 'H']
    wiersze = [1, 2, 3, 4, 5, 6, 7, 8]
    # eby byo jak na szachownicy

    plt.figure(figsize=(7, 7))

    ax = sns.heatmap(
        counts,
        cmap="Greens",
        linewidths=1.2,
        linecolor="black",
        square=True,
        cbar=True,
        cbar_kws={'label': 'Ile razy wystpuje pozycja'},
        annot=True,
        xticklabels=kolumny,
        yticklabels=wiersze
    )

    cbar = ax.collections[0].colorbar
    cbar.outline.set_visible(True)
    cbar.outline.set_linewidth(1.2)
    cbar.outline.set_edgecolor("black")

    plt.title(title, fontweight='bold')
    plt.xlabel("Kolumna")
    plt.ylabel("Wiersz")
    
    plt.gca().invert_yaxis()
    # eby byo jak na szachownicy
    
    plt.show()

config_metryki = [
    ("euclidean", "Euklidesowa", "#a1d99b"),

for metryka, nazwa, kolor in config_metryki:

    corr = df_bez_remisow[metryka].corr(df_bez_remisow["white-depth-of-win"])
    tekst = f"{nazwa} (r = {corr:.2f})"
    
    x_data = df_bez_remisow[metryka]

    sns.lineplot(
        x=x_data,
        y=df_bez_remisow["white-depth-of-win"],
        label=tekst,
        color=kolor,
        linewidth=2.5,
        errorbar=None
    )

plt.title("Liczba ruchw potrzebna do wygranej vs dystans midzy krlami", fontweight='bold')

Odpowied:

Analiza korelacji wskazuje, e metryka Manhattan (r = 0.30) pozwala najlepiej oszacowa liczb ruchw do zakoczenia gry, osigajc odrobin wyszy wynik ni metryka Euklidesowa i metryka Czebyszewa. Wykres potwierdza, e dla metryki Manhattan wzrost redniej liczby ruchw wraz z dystansem jest najbardziej stabilny. Poza tym, metryka Manhattan wydaje si by najbardziej intuicyjna dla szachownicy.
